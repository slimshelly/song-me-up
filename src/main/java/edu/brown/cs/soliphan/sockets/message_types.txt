Each has two keys: type, payload

type: CONNECT, payload: {
	id: Spotify's user id
	}

type: VOTESONG, 
payload: {
	id: Spotify's user id,
	song_id: Spotify id for song,
	vote: boolean. true for upvote, false for downvote.
	}

type: ADDSONG,
payload: {
	id: Spotify's user id,
	song_id: Spotify id for song
	}
